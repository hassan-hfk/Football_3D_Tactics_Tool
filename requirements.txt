# Covers: FastAPI server, YOLO detection + BoT-SORT tracking, PnLCalib
# homography (HRNet-W48), ViTPose-B (ONNX), GVHMR SMPL-X mesh lifting,
# and ffmpeg-based MP4 export.
#
# Install:
#   pip install -r requirements.txt
#
# Do NOT use the requirements.txt files inside PnLCalib/, MotionBERT/, or
# research/GVHMR/ — those pin Linux/CUDA-only wheels and will fail on
# Windows/macOS.
#
# pytorch3d is NOT listed here because it cannot be pip-installed on most
# setups. See SETUP.md for the full build instructions.

torch
torchvision

imageio-ffmpeg         # bundled ffmpeg binary used by /transcode-to-mp4 endpoint
ffmpeg-python          # GVHMR imports this (NOT the bare `ffmpeg` package)

hydra-core             # GVHMR's config system
omegaconf              # dependency of hydra-core
lightning              # GVHMR wraps its model as a LightningModule
pytorch-lightning      # alias kept for older import paths
timm                   # HMR2 ViT backbone
colorlog               # logging dependency pulled in by GVHMR / hydra
hydra-zen              # used by some GVHMR config helpers
wis3d                  # lightweight 3D visualisation used by GVHMR utilities
chumpy                 # SMPL serialisation legacy dependency
